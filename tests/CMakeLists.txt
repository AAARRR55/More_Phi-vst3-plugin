# MorphSnap — Test Suite (morphsnap:: API, Catch2 v3)
# Only compiles real, existing test sources.

cmake_minimum_required(VERSION 3.24)
enable_testing()

# ── Catch2 v3 ──────────────────────────────────────────────────────────────────
if(NOT TARGET Catch2::Catch2WithMain)
    include(FetchContent)
    FetchContent_Declare(
        Catch2
        GIT_REPOSITORY https://github.com/catchorg/Catch2.git
        GIT_TAG        v3.4.0
        GIT_SHALLOW    TRUE
    )
    FetchContent_MakeAvailable(Catch2)
endif()

# ── Unit test executable ────────────────────────────────────────────────────────
add_executable(MorphSnapTests
    Unit/test_morphsnap_unit.cpp
    Unit/TestPhysicsAndGenetic.cpp
    Unit/TestSidechainTrigger.cpp
    Integration/TestPluginLifecycle.cpp
)

# Link the compiled plugin objects in (shared) so the test binary has access
# to ParameterState, LockFreeQueue, InterpolationEngine, SnapshotBank
target_link_libraries(MorphSnapTests PRIVATE
    Catch2::Catch2WithMain
    MorphSnap                 # JUCE shared-code target generated by juce_add_plugin
    juce::juce_core           # propagates JUCE include paths to test translation units
    juce::juce_audio_processors
    nlohmann_json::nlohmann_json
)

target_include_directories(MorphSnapTests PRIVATE
    ${CMAKE_SOURCE_DIR}/src
)

target_compile_definitions(MorphSnapTests PRIVATE
    MORPHSNAP_TEST_MODE=1
    JUCE_UNIT_TESTS=1
    JUCE_STANDALONE_APPLICATION=0
)

# ── Sanitizers (opt-in via parent MORPHSNAP_ENABLE_SANITIZERS) ─────────────────
if(MORPHSNAP_ENABLE_SANITIZERS AND (CMAKE_CXX_COMPILER_ID MATCHES "Clang|GNU"))
    target_compile_options(MorphSnapTests PRIVATE
        -fsanitize=address,undefined
        -fno-omit-frame-pointer
    )
    target_link_options(MorphSnapTests PRIVATE
        -fsanitize=address,undefined
    )
endif()

# ── CTest registration ─────────────────────────────────────────────────────────
include(Catch)
catch_discover_tests(MorphSnapTests
    TEST_SPEC "*"
)

# ── Benchmark executable (opt-in) ──────────────────────────────────────────────
option(MORPHSNAP_BUILD_BENCHMARKS "Build performance benchmark suite" ON)
if(MORPHSNAP_BUILD_BENCHMARKS)
    add_executable(MorphSnapBenchmarks
        Performance/BenchmarkSuite.cpp
    )

    target_link_libraries(MorphSnapBenchmarks PRIVATE
        MorphSnap
        juce::juce_core
        juce::juce_audio_processors
    )

    target_include_directories(MorphSnapBenchmarks PRIVATE
        ${CMAKE_SOURCE_DIR}/src
    )

    target_compile_definitions(MorphSnapBenchmarks PRIVATE
        MORPHSNAP_TEST_MODE=1
        JUCE_STANDALONE_APPLICATION=0
    )
endif()

# VST3 validation (optional, needs vst3_validator on PATH)
if(WIN32)
    find_program(VST3_VALIDATOR_EXE vst3_validator)
    if(VST3_VALIDATOR_EXE)
        add_test(NAME VST3Validation
            COMMAND ${VST3_VALIDATOR_EXE} -r $<TARGET_FILE:MorphSnap_VST3>
        )
        set_tests_properties(VST3Validation PROPERTIES TIMEOUT 120 LABELS validation)
    endif()
endif()

if(APPLE)
    find_program(AUVAL_EXE auval)
    if(AUVAL_EXE)
        add_test(NAME AUValidation
            COMMAND ${AUVAL_EXE} -v aufx Mrph Mrph
        )
        set_tests_properties(AUValidation PROPERTIES TIMEOUT 120 LABELS validation)
    endif()
endif()
